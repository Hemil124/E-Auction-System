CREATE TABLE `e-auction`.`tblusers` (`Id` INT NOT NULL AUTO_INCREMENT , `FirstName` VARCHAR(20) NOT NULL , `LastName` VARCHAR(20) NOT NULL , `MobileNo` VARCHAR(10) NOT NULL , `Email` VARCHAR(200) NULL , `Address` VARCHAR(500) NOT NULL , `DateofBirth` DATE NOT NULL , `Password` VARCHAR(20) NOT NULL , `Role` VARCHAR(6) NOT NULL , `UserImage` BLOB NOT NULL , PRIMARY KEY (`Id`), UNIQUE (`Email`)) ENGINE = InnoDB;
